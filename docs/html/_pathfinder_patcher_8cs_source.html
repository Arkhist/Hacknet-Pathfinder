436  internal static AssemblyDefinition LoadAssembly(string fileName, ReaderParameters parameters = null)
437  {
438  parameters = parameters ?? new ReaderParameters(ReadingMode.Deferred);
439  if (string.IsNullOrEmpty(fileName))
440  {
441  throw new ArgumentException("fileName is null/empty");
442  }
443  Stream stream = new FileStream(fileName, FileMode.Open, parameters.ReadWrite ? FileAccess.ReadWrite : FileAccess.Read, FileShare.Read);
444  if (parameters.InMemory)
445  {
446  var memoryStream = new MemoryStream(stream.CanSeek ? ((int)stream.Length) : 0);
447  using (stream)
448  {
449  stream.CopyTo(memoryStream);
450  }
451  memoryStream.Position = 0L;
452  stream = memoryStream;
453  }
454  ModuleDefinition result;
455  try
456  {
457  result = ModuleDefinition.ReadModule(stream, parameters);
458  }
459  catch (Exception)

461  stream.Dispose();
462  throw;
463  }
464  return result.Assembly;
